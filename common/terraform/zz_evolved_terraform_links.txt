terraform_link = {
	from = "pc_barren"
	to = "pc_desert"

	resources = {
		category = terraforming

		cost = {
			energy = 10000
		}
	}

	duration = 7200

	potential = {
		exists = owner
		from = { has_modifier = pm_tec_desiccated_planet }
	}

	condition = {
		has_technology = "tech_climate_restoration"
		custom_tooltip = {
			fail_text = tec_machine_anglers_cannot_restore_desiccated
			NOT = { has_valid_civic = civic_tec_machine_anglers }
		}
	}

	effect = {
		from = { remove_modifier = pm_tec_desiccated_planet }
	}

	ai_weight = {
		weight = 5
	}
}

terraform_link = {
	from = "pc_barren"
	to = "pc_machine"

	resources = {
		category = terraforming

		cost = {
			energy = @machineCost
		}
	}

	duration = @machineTime

	potential = {
		exists = owner
		has_ascension_perk = ap_machine_worlds
		from = { has_modifier = pm_tec_desiccated_planet }
	}

	effect = {
		from = {
			reroll_deposits = yes
			clear_blockers = yes
			remove_modifier = pm_tec_desiccated_planet
			add_deposit = d_tec_desiccators_deposit
		}
	}

	ai_weight = {
		weight = 200
		#It's a barren world!
	}
}
