building_military_academy = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	position_priority = 105		# Army

	category = army

	prerequisites = {
		"tech_centralized_command"
	}
	
	show_tech_unlock_if = {
		NOT = { tec_is_reanimator_empire = yes }
		tec_is_biohive = no
	}

	potential = {
		exists = owner
		owner = { NOT = { tec_is_reanimator_empire = yes } }
	}

	allow = {
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
				has_building = building_tec_bastion
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				tec_is_reanimator_empire = yes
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
							has_building = building_tec_bastion
						}
					}
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
		building_tec_biohive_military
	}
	

	army_modifier = {
		army_starting_experience_add = 100
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @vanilla2jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @vanilla2jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}

		# Special resource civics
		cost = {
			trigger = {
				owner = { tec_is_bioarch_empire = yes }
			}
			minerals = @b1_minus_minerals
			food = @b1_half_minerals
		}
		upkeep = {
			trigger = {
				owner = { tec_is_bioarch_empire = yes }
			}
			energy = @b1_minus_upkeep
			food = @b1_half_upkeep
		}
		cost = {
			trigger = {
				owner = { has_valid_civic = civic_tec_machine_digitalization }
			}
			minerals = @b1_minus_minerals
			energy = @b1_half_minerals
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}

}
