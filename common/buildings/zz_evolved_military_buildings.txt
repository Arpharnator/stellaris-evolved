		script = jobs/inline_evolved_soldier

	}

	inline_script = {
		script = jobs/inline_evolved_warrior_drone
		count = @vanilla2jobs
		mult = 1
		display = with_desc
		trigger = ""

			trigger = {
				exists = owner

					has_valid_civic = civic_tec_regular_stratocracy
