building_astral_siphon_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = research

	potential = {
		owner = {
			check_variable = {
				which = astral_siphon_unlocked
				value >= 1
			}
		}
	}

	resources = {
		category = planet_buildings
		inline_script = {
			script = buildings/resources/tec_resource_cost
			tier = 1
		}
	}

	inline_script = buildings/convert_to/tec_research_1

	country_modifier = {
		country_resource_max_astral_threads_add = 100
	}

	inline_script = {
		script = jobs/tec_astral_researcher
		count = @b1_jobs
		mult = 1
		display = with_desc
		trigger = ""
		default_job = astral_researcher
	}


	inline_script = {
		script = buildings/effects/type/tec_research_lab
		tier = 1
		type = all
	}


	upgrades = {
		building_astral_siphon_2
	}
}
