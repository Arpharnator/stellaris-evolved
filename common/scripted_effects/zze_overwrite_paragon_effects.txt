country_create_gestalt_councilor = {
	set_country_flag = tec_force_councilor_traits
	create_leader = {
		class = $CLASS$
		species = this
		name = $COUNCIL_POS$
		skill = 1
		gender = indeterminable
		event_leader = yes
		skip_background_generation = yes
		can_manually_change_location = no
		hide_leader = yes
		custom_description = $DESC$
		randomize_traits = no
		background_ethic = ethic_gestalt_consciousness
		effect = {
			save_event_target_as = gestalt_councilor
		}
	}
	event_target:gestalt_councilor = {
		set_owner = prev
		change_leader_portrait = $COUNCIL_POS$_1
		set_council_position = $COUNCIL_POS$
		if = {
			limit = {
				has_paragon_dlc = yes
			}
			assign_node_subclass_traits = yes
		}
		#tec_leader_add_councilor_trait = yes
		tec_leader_handle_starting_modifiers = {	# Evolved
			from = prev
		}
		set_immortal = yes
		set_leader_flag = leader_death_events_blocked
	}
	remove_country_flag = tec_force_councilor_traits
}


assign_restored_node = {
	event_target:restored_node = {
		set_owner = prev
		change_leader_portrait = $COUNCIL_POS$_1
		set_council_position = $COUNCIL_POS$
		set_immortal = yes
		set_leader_flag = leader_death_events_blocked
		if = {
			limit = {
				has_paragon_dlc = yes
			}
			assign_node_subclass_traits = yes
		}
	}
}

add_bonus_skills_effect = {
	tec_leader_handle_starting_modifiers = {
		from = owner
	}
}
