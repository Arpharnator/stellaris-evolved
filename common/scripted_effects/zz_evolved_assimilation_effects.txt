	tec_set_common_assimilation_flags = {
		optimize_memory
		set_timed_species_flag = {
			flag = assimilation_species_of_species_@prev
			days = 30
		}

		}

	tec_assimilation_remove_hivemind_script = {
		optimize_memory
		modify_species = {
			species = this
			remove_trait = trait_self_modified
			remove_trait = trait_hive_mind
			remove_trait = trait_tec_hive_psionic_entanglement
			remove_trait = trait_tec_hive_psionic_resonance
			change_scoped_species = no
			effect = {
				tec_set_common_assimilation_flags = yes
				save_event_target_as = tec_result_species
			}
		}
	}

# Assimilation Effect
# ------------------------------------------------------------
	tec_pop_assimilation_effect = {
		optimize_memory
		
		random_owned_pop = {
			limit = {
				has_citizenship_type = {
					type = citizenship_assimilation
					country = owner
				}
			}
			if = {	# Randomize the fail chance TO DO
				limit = { 
					species = {
						NOR = {
							has_trait = trait_mechanical
							has_trait = trait_machine_unit
							has_trait = @trait_tec_bioservant
						}
						OR = {
							has_living_standard = {
								type = living_standard_tec_assimilation_cybernetic_t2_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_synthetic_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_synthetic_total_machine
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_genetic_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_biopsionic_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_replicant_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_biological_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_bionic_total
								country = owner
							}
							has_living_standard = {
								type = living_standard_tec_assimilation_ethereal_total
								country = owner
							}
						}
					}
				}
				tec_assimilation_fail_assign_percent = yes
			}
