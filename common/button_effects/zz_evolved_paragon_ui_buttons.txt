tec_cui_aiuth_fragment_commander_button_effect = {
	potential = {
		exists = FROM
		FROM = {
			tec_is_aiauth_empire = yes
			has_global_flag = tec_allow_cui
		}
	}
	allow = {
		custom_tooltip = tec_cui_aiuth_fragment_commander_effect
		custom_tooltip = tec_cui_aiauth_fragmenting_cost_effect
		FROM = {
			tec_country_aiauth_has_enough_unity_to_fragment = yes
			custom_tooltip = {
				fail_text = tec_cui_aiauth_fragmenting_node
				NOT = { has_country_flag = tec_fragmenting_node }
			}
		}
	}
	effect = {
		print_scope_effect = yes
		custom_tooltip = tec_cui_aiuth_fragment_commander_effect
		custom_tooltip = tec_cui_aiauth_fragmenting_cost_effect
		hidden_effect = {
			from = {
				tec_country_fragment_aiauth_node_cost = yes
				set_country_flag = tec_fragmenting_commander
				tec_country_create_aiauth_councilor = {
					class = commander
					name = councilor_tec_aiauth_fragment_commander
					desc = councilor_tec_aiauth_fragment_commander_desc
				}
				country_event = {
					id = tec_aiauth.20
				}
			}
		}
	}
}
