tech_habitat_2 = {
	cost = @tier4cost2
	area = engineering
	category = { voidcraft }
	tier = 4
	prerequisites = { "tech_habitat_1" }
	weight = @tier4weight3
	is_rare = yes

	potential = {
		OR = {
			has_origin = origin_void_dwellers
			host_has_dlc = "Utopia"
		}
	}

	prereqfor_desc = {
		ship = {
			title = "allow_habitat_upgrade"
			desc = "tech_habitat_2_effect"
		}
	}

	weight_modifier = {
		modifier = {
			factor = 0
			NOT = {
				any_owned_planet = {
					is_planet_class = pc_habitat
				}
			}
		}
		modifier = {
			factor = 10
			has_origin = origin_void_dwellers
		}
	}

	ai_weight = {
		modifier = {
			factor = @tec_tech_important_ai_tech_factor
			has_origin = origin_void_dwellers
			years_passed > 25
		}
		complex_trigger_modifier = {
			trigger = count_owned_planet
			parameters = {
				limit = {
					is_planet_class = pc_habitat
					tec_is_upgraded_habitat = no
				}
			}
			mode = add
		}
	}
}
