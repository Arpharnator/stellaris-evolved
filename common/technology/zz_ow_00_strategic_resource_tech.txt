	cost = @tier5cost1
	area = engineering
	tier = 5
	category = { materials }
	prerequisites = { "tech_space_construction" }
	weight = @tier5weight1

	weight_modifier = {
		factor = 0.5

			}
		}
	}

			factor = 20
			has_country_flag = dragon_breeders #so they can recruit Fledgeling Dragons
		}
	}
}
