	tec_cooperative_ethics_from_relations = {
		[[CACHED]
		modifier = {
			is_variable_set = tec_cache_tec_cooperative_ethics_from_relations
			add = tec_cache_tec_cooperative_ethics_from_relations # cache_variable
		}
		]
		[[!CACHED]
		base = 1
		complex_trigger_modifier = {
			trigger = count_relation
			parameters = {
				limit = {
					OR = {
						is_in_federation_with = root
						has_defensive_pact = root
						has_commercial_pact = root
					}
					is_cooperative = yes
				}
			}
			mode = add
			mult = 0.3
		}
		complex_trigger_modifier = {
			trigger = count_relation
			parameters = {
				limit = {
					is_subject = no
					has_migration_access = root
					is_cooperative = yes
				}
			}
			mode = add
			mult = 0.5
		}
		]
	}
