# NOTE: These are only to be used under [[NEVER] ... ] syntax in script values, to force parameter type hinting

tec_type_hint_job = {
    modifier = {
        add = 0
        has_job = $PARAM$
    }
}


tec_type_hint_district = {
    modifier = {
        add = 0
        has_district = $PARAM$
    }
}

tec_type_hint_building = {
    modifier = {
        add = 0
        has_building = $PARAM$
    }
}

tec_type_hint_scope = {
    modifier = {
        add = 0
        exists = $PARAM$
    }
}

tec_type_hint_bool = {
    modifier = {
        add = 0
        always = $PARAM$
    }
}

tec_type_hint_resource = {
    modifier = {
        add = 0
        has_resource = $PARAM$
    }
}

tec_type_hint_number_or_variable = {
    add = $PARAM$
}

tec_type_hint_script_value = {
    add = value:$PARAM$
}


tec_type_hint_trait = {
    modifier = {
        add = 0
        has_trait = $PARAM$
    }
}

tec_type_hint_pop_category = {
    modifier = {
        add = 0
        is_pop_category = $PARAM$
    }
}

tec_type_hint_research_category = {
    modifier = {
        add = 0
        can_copy_random_tech_from = {
            category = $PARAM$
            who = this
        }
    }
}
