tec_count_positive_traits = {
	inline_script = {
		script = tec_type_hint
		[[BASE] float = $BASE$ ]
		[[BASE_VALUE] script_value = $BASE_VALUE$ ]
	}

	BASE = 0
	[[BASE]add = $BASE$]
	[[BASE_VALUE]add = value:$BASE_VALUE$]

	complex_trigger_modifier = {
		trigger = num_positive_traits
		trigger_scope = from
		mode = add
	}
}
