districts_value = {
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = $TYPE$
		}
		[[FACTOR]multiplier = $FACTOR$]
		mode = add
	}
}

		trigger = num_buildings
		parameters = {
			type = $TYPE$
		}
		[[FACTOR]multiplier = $FACTOR$]
		mode = add
	}
}
