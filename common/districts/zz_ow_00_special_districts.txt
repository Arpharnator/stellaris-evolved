@lathe_alloy_cost = 1000
@lathe_logic_cost = 500
@lathe_upkeep = 50

district_orders_demesne = {
	base_buildtime = 240
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = habitat
		OR = {
			has_planet_flag = toxic_knights_habitat
			has_building = building_order_castle
		}
	}

	potential = {
		uses_district_set = habitat
		exists = owner
		owner = { has_origin = origin_toxic_knights }
		OR = {
			has_planet_flag = toxic_knights_habitat
			has_building = building_order_castle
		}
	}

	conversion_ratio = 0.5
	inline_script = districts/convert_to/tec_housing_districts

	resources = {
		category = planet_districts_orders_demesne

		inline_script = {
			script = districts/resources/tec_district_resource_cost
			upkeep_trigger = "
				always = yes
			"
			cost = @tec_district_hab_cost
			upkeep = @tec_district_hab_upkeep
		}
		upkeep = {
			alloys = @tec_district_hab_upkeep_alloys
		}
	}

	planet_modifier = {
		planet_max_buildings_add = 0.50
		planet_housing_add = 6
		job_squire_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			owner = {
				is_void_dweller_empire = yes
			}
		}
		job_squire_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			has_deposit = d_dimensional_manipulation_device
			exists = owner
			NOT = {
				owner = { has_deficit = sr_dark_matter }
			}
		}
		planet_max_buildings_add = 0.5
		planet_housing_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			has_deposit = d_dimensional_manipulation_device
			exists = owner
			owner = { has_deficit = sr_dark_matter }
		}
		planet_housing_add = -1
	}

	# Arcology habitat
		triggered_planet_modifier = {
			potential = {
				tec_is_arcology_habitat = yes
			}
			planet_housing_add = 2
			job_squire_add = 2
		}
		triggered_planet_modifier = {
			potential = {
				tec_is_arcology_habitat = yes
				has_deposit = d_dimensional_manipulation_device
				exists = owner
				NOT = {
					owner = { has_deficit = sr_dark_matter }
				}
			}
			planet_max_buildings_add = 1
			planet_housing_add = 2
		}
		triggered_planet_modifier = {
			potential = {
				tec_is_arcology_habitat = yes
				has_deposit = d_dimensional_manipulation_device
				exists = owner
				owner = { has_deficit = sr_dark_matter }
			}
			planet_housing_add = -2
		}
	# /

	triggered_desc = {
		text = job_squire_effect_desc
	}
}
