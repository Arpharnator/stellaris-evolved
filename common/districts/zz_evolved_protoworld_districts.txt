		category = planet_districts_cities

			minerals = @proto_citycost
		}
		upkeep = {
			energy = @proto_upkeep
		}
	}
	
	# Modifiers
		# Standard
			planet_modifier = {
				planet_housing_add = 2
			}

			script = districts/inline_evolved_habitation_housing_modifiers

			trigger = ""
		}

		inline_script = districts/effects/inline_evolved_tec_artifice_2

	# /
