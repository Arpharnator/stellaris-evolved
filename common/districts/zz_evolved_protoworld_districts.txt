# --------------- Habitation
district_tec_proto_city = {
	icon = district_city
	base_buildtime = 480
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		from = {
			tec_is_biomorph = no
		}
		uses_district_set = tec_protoworld		
	}
	
	potential = {
		exists = owner
		owner = {
			tec_is_biomorph = no
		}
		uses_district_set = tec_protoworld	
	}
	
	allow = {
		NOT = {
			has_modifier = resort_colony 
		}
		NOT = {
			has_modifier = slave_colony 
		}
	}

	conversion_ratio = 0.5
	convert_to = {
		district_city
		district_arcology_housing
		district_tec_arcology_trade
		district_nexus
		district_hive
		district_tec_eco_arcology
		#district_tec_proto_hive
	}
	
	resources = {
		category = planet_districts_cities
		cost = {
			minerals = @proto_citycost
		}
		upkeep = {
			energy = @proto_upkeep
		}
	}
	
	# Modifiers
		# Standard
			planet_modifier = {
				planet_housing_add = 2
			}
		# /
		# Technologies
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						has_technology = tech_housing_1 
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						has_technology = tech_housing_2 
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}
		# /
		# Traditions
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						has_active_tradition = tr_prosperity_public_works 
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}

			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						is_gestalt = no
						OR = {
							has_tradition = tr_tec_artifice_2
							has_active_tradition = tr_tec_artifice_2_gestalt
						}
					}
				}
				planet_amenities_add = 1
			}	
			triggered_planet_modifier = {
				potential = {
					exists = owner
					owner = {
						is_gestalt = yes
						OR = {
							has_tradition = tr_tec_artifice_2
							has_active_tradition = tr_tec_artifice_2_gestalt
						}
					}
				}
				planet_amenities_no_happiness_add = 1
			}
		# /
	# /
}
