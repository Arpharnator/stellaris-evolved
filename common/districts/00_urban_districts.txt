# --------------- Habitation
district_city = {
	base_buildtime = 480
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes
	
	show_on_uncolonized = {
		exists = from
		from = { is_regular_empire = yes }
		tec_has_standard_habitation_district = yes
	}
	
	potential = {
		OR = {
			NOT = { exists = owner } #this condition is needed for galaxy generation
			AND = {
				exists = owner
				owner = {
					OR = {
						is_regular_empire = yes
						is_country_type = primitive
					}
				}
			}
		}
		tec_has_standard_habitation_district = yes
	}
	
	allow = {
		NOT = { has_modifier = resort_colony }

	conversion_ratio = 1

	resources = {
		category = planet_districts_cities
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
	}
	
	# Modifiers
		# Standard
			planet_modifier = {
				planet_housing_add = 5
				planet_max_buildings_add = 1
			}
		# /
		# Technologies
			triggered_planet_modifier = {	# Housing t1
				potential = {
					exists = owner
					owner = {
						has_technology = tech_housing_1 
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}
			triggered_planet_modifier = {	# Housing t2
				potential = {
					exists = owner
					owner = {
						has_technology = tech_housing_2
						tec_is_agrarian_idyll_empire = no
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}
		# /
		# Traditions
			triggered_planet_modifier = {	# Prosperity
				potential = {
					exists = owner
					owner = { has_active_tradition = tr_prosperity_public_works }
				}
				modifier = {
					planet_housing_add = 1
				}
			}
			triggered_planet_modifier = {	# Mercantile
				potential = {
					exists = owner
					owner = { has_active_tradition = tr_mercantile_trickle_up_economics }
				}
				modifier = {
					job_clerk_add = 1
				}
			}
			triggered_planet_modifier = {	# Artifice
				potential = {
					exists = owner
					owner = {
						OR = {
							has_tradition = tr_tec_artifice_2
							has_active_tradition = tr_tec_artifice_2_gestalt
						}
					}
				}
				planet_amenities_add = 2
			}
		# /
		# Civics
			triggered_planet_modifier = {	# Idyll
				potential = {
					exists = owner
					owner = {
						tec_is_agrarian_idyll_empire = yes
					}
				}
				modifier = {
					planet_housing_add = -2
				}
			}
			triggered_planet_modifier = {	# Keepers
				potential = {
					exists = owner
					owner = {
						OR = { has_civic = civic_tec_eco_keepers }
					}
				}
				modifier = {
					planet_housing_add = -1
				}
			}
			triggered_planet_modifier = {	# Tribalism
				potential = {
					exists = owner
					owner = {
						has_valid_civic = civic_tec_tribalism
					}
				}
				modifier = {
					planet_housing_add = -2
				}
			}
			triggered_planet_modifier = {	# Social prosperity
				potential = {
					exists = owner
					owner = {
						OR = { has_valid_civic = civic_tec_social_prosperity }
					}
				}
				modifier = {
					planet_housing_add = 1
				}
			}
			triggered_planet_modifier = {	# Consumerism
				potential = {
					exists = owner
					owner = {
						has_valid_civic = civic_tec_consumerism
					}
				}
				modifier = {
					trade_value_add = 2
				}
			}
		# /
	# /

	# Jobs
		# Base
			triggered_planet_modifier = {		# Clerks
				potential = {
					exists = owner
					owner = {
						is_regular_empire = yes
					}
				}
				modifier = {
					job_clerk_add = 1
