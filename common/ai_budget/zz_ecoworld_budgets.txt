	resource = minerals
	type = expenditure
	category = decisions

	potential = {
		NOR = {
			has_ethic = ethic_fanatic_industrial
			has_valid_civic = civic_hive_devouring_swarm
		}
		OR = {
			is_ecologist = yes
			has_ascension_perk = ap_mastery_of_nature
		}
		any_owned_planet = {
			tec_ecoworld_project_potential = yes

				num_pops >= 70

			}
