energy_expenditure_deposit_blockers = {
	resource = energy
	type = expenditure
	category = deposit_blockers

	potential = {
		OR = {
			AND = {
				has_resource = {
					type = energy
					amount > 1000
				}
				any_owned_planet = {
					num_uncleared_blockers > 0
				}
			}
			any_owned_planet = {
				OR = {
					has_deposit = d_unsupervised_settlement
					has_deposit = d_collapsed_burrows
					has_deposit = d_decrepit_dwellings
				}
			}
		}
	}

	weight = {
		weight = 0.3

		modifier = {
			factor = 5

			# if we have any planet that is currently completely blocked
			any_owned_planet = {
				num_uncleared_blockers > 0
				num_free_districts = { type = any value < 1 }
			}
		}
	}

	# base remove blocker cost
	desired_min = {
		base = 500
	}

	desired_max = {
		base = 1000
	}
}
