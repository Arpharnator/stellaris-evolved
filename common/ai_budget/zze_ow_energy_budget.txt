energy_expenditure_deposit_blockers = {
	resource = energy
	type = expenditure
	category = deposit_blockers

	potential = {
		OR = {
			AND = {
				has_resource = {
					type = energy
					amount > 1000
				}
				any_owned_planet = {
					num_uncleared_blockers > 0
				}
			}
			any_owned_planet = {
				OR = {
					has_deposit = d_unsupervised_settlement
					has_deposit = d_collapsed_burrows
					has_deposit = d_decrepit_dwellings
				}
			}
		}
	}

	weight = {
		weight = 0.3

		modifier = {
			factor = 5

			# if we have any planet that is currently completely blocked
			any_owned_planet = {
				num_uncleared_blockers > 0
				num_free_districts = { type = any value < 1 }
			}
		}
	}

	# base remove blocker cost
	desired_min = {
		base = 500
	}

	desired_max = {
		base = 1000
	}
}

energy_expenditure_colonies = {
	resource = energy
	type = expenditure
	category = colonies

	potential = {
		ai_colonize_plans > 0
		tec_has_private_colony_ships = yes
	}

	# if we can colonize with energy, we should
	weight = {
		weight = 1.5
	}

	desired_min = {
		base = @colony_cost_high
		modifier = {
			mult = 0.5
			has_active_tradition = tr_tec_outreach_final_3
		}
	}

	desired_max = {
		base = @colony_cost_high_max
		modifier = {
			mult = 0.5
			has_active_tradition = tr_tec_outreach_final_3
		}
	}
}
