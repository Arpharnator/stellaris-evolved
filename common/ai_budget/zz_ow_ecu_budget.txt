minerals_expenditure_decisions = {
	resource = minerals
	type = expenditure
	category = decisions

	potential = {
		OR = {
			AND = {
				has_technology = tech_housing_2
				any_owned_planet = {
					tec_arcology_project_relic_potential = yes
					tec_arcology_project_relic_allow = yes
				}
			}
			AND = {
				has_ascension_perk = ap_arcology_project
				any_owned_planet = {
					tec_arcology_project_potential = yes
					tec_arcology_project_allow = yes
					OR = {
						num_pops >= 70
						is_capital = yes
					}
				}
			}
		}
		# only consider these kinds of decisions when "wealthy"
		has_resource = {
			type = minerals
			amount > 25000
		}
	}

	weight = {
		weight = 0.20
	}

	desired_min = {
		base = 20000
	}
	desired_max = {
		base = 20000
	}
}
