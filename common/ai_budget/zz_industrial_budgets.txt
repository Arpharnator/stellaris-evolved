
tec_expenditure_energy_decisions_industrial_reengineer = {
	resource = energy
	type = expenditure
	category = decisions

	potential = {
		tec_is_engineer_empire = yes
		any_owned_planet = {
			tec_is_space_arcology = yes
			# Exception - Shattered rings can't be upgraded
				NOT = {
					uses_district_set = shattered_ring_world
				}
			# /
			NOT = {
				has_planet_flag = tec_enhanced_by_terraformers
			}
			OR = {
				num_pops > 25
				is_capital = yes
				AND = {
					exists = owner
					owner = { has_origin = origin_void_dwellers }
					tec_is_habitat = yes
					has_planet_flag = advanced_habitat
				}
			}
		}
		
		has_resource = {
			type = energy
			amount > 10000
		}
	}

	weight = {
		weight = 0.2
	}
	desired_min = {
		base = 7500
	}
	desired_max = {
		base = 7500
	}
}
