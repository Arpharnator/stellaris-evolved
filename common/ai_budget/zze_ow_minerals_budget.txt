minerals_expenditure_stations_low = {
	resource = minerals
	type = expenditure
	category = stations

	potential = {
		if = {
			limit = {
				tec_is_biohive = yes
			}
			NOT = {	# Doubledip
				has_modifier = modifier_tec_hidden_organicarch_modifier
			}
		}
		else = {
			always = yes
		}
		resource_stockpile_compare = { resource = minerals value < 1000 }
	}

	weight = {
		weight = 0.2
	}
}
