minerals_expenditure_stations_low = {
	resource = minerals
	type = expenditure
	category = stations

	potential = {
		if = {
			limit = {
				tec_is_biohive = yes
			}
			NOT = {	# Doubledip
				has_modifier = modifier_tec_hidden_organicarch_modifier
			}
		}
		else = {
			always = yes
		}
		resource_stockpile_compare = { resource = minerals value < 1000 }
	}

	weight = {
		weight = 0.2
	}
}

minerals_expenditure_colonies_lithoid_expand = {
	resource = minerals
	type = expenditure
	category = colonies

	potential = {
		is_robot_empire = no
		is_lithoid_empire = yes
		NOT = {
			has_origin = origin_synthetic_fertility
		}
		has_resource = { type = minerals amount > 400 }
		has_monthly_income = {
			resource = minerals
			value > 0
		}
	}

	weight = {
		weight = 0.8
	}

	desired_min = {
		base = 0
		modifier = {
			add = @colony_cost_base
			is_hive_empire = no
		}

		modifier = {
			add = @colony_cost_high
			is_hive_empire = yes
			NOT = { has_origin = origin_lithoid }
		}
		modifier = {
			mult = 0
			has_active_tradition = tr_tec_outreach_final_3
		}
		modifier = {
			add = @colony_cost_high
			has_origin = origin_lithoid
		}
		modifier = {
			mult = 0.5
			has_active_tradition = tr_tec_outreach_final_3
		}
	}

	desired_max = {
		base = 0
		modifier = {
			add = @colony_cost_base_max
			is_hive_empire = no
		}
		modifier = {
			add = @colony_cost_high_max
			is_hive_empire = yes
			NOT = { has_origin = origin_lithoid }
		}
		modifier = {
			mult = 0
			has_active_tradition = tr_tec_outreach_final_3
		}
		modifier = {
			add = @colony_cost_high_max
			OR = {
				is_hive_empire = yes
				has_origin = origin_lithoid
			}
		}
		modifier = {
			mult = 0.5
			has_active_tradition = tr_tec_outreach_final_3
		}
	}
}


minerals_expenditure_deposit_blockers = {
	resource = minerals
	type = expenditure
	category = deposit_blockers

	potential = {
		any_owned_planet = {
			OR = {
				has_deposit = d_ruined_arcology
				has_deposit = d_resource_consolidation_1
				has_deposit = d_tec_machine_rubble_big
				AND = {
					OR = {
						has_deposit = d_buried_lithoids
						has_deposit = d_hibernating_lithoids
					}
					is_homeworld = yes
				}
			}
		}
	}

	weight = {
		weight = 0.3
		modifier = {
			factor = 5

			# if we have any planet that is currently completely blocked
			any_owned_planet = {
				OR = {
					has_deposit = d_ruined_arcology
					AND = {
						OR = {
							has_deposit = d_buried_lithoids
							has_deposit = d_hibernating_lithoids
						}
						is_homeworld = yes
					}
				}
				num_free_districts = { type = any value < 1 }
			}
		}
	}

	# base remove blocker cost
	desired_min = {
		base = 250
	}

	desired_max = {
		base = 1500
	}
}
