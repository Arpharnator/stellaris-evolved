	tec_gene_automod_checker = {
		hidden_trigger = {
			exists = from
			[[T1]
			# T1 should not trigger if the t2 or t3 is available
			switch = {
				trigger = has_trait
				trait_tec_gene_automodding_t1 = {
					always = yes
				}
				trait_tec_gene_automodding_t2 = {
					from = {
						NOR = {
							has_country_flag = unlocked_$TRAIT$_t2
							has_country_flag = unlocked_$TRAIT$_t3
						}
					}
				}
				trait_tec_gene_automodding_t3 = {
					from = {
						NOR = {
							has_country_flag = unlocked_$TRAIT$_t2
							has_country_flag = unlocked_$TRAIT$_t3
						}
					}
				}
			}
			]
			[[!T1]]
			[[T2]
			# T2 should appear for t2 or t3
			switch = {
				trigger = has_trait
				trait_tec_gene_automodding_t1 = {
					always = no
				}
				trait_tec_gene_automodding_t2 = {
					from = {
						NOT = {
							has_country_flag = unlocked_$TRAIT$_t3
						}
					}
				}
				trait_tec_gene_automodding_t3 = {
					from = {
						NOT = {
							has_country_flag = unlocked_$TRAIT$_t3
						}
					}
				}
			}
			]
			[[!T2]]
			[[T3]
			# T3 should not appear for t1 or t2 automodding
			switch = {
				trigger = has_trait
				trait_tec_gene_automodding_t1 = {
					always = no
				}
				trait_tec_gene_automodding_t2 = {
					always = no
				}
				trait_tec_gene_automodding_t3 = {
					always = yes
				}
			}
			]
			[[!T3]]
		}
	}
