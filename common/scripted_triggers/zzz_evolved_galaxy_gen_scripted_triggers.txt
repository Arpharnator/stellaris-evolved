	tec_has_orbital_physics_deposit = {
		OR = {
			has_resource = physics_research

			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_physics_deposit
			}
		}
	}

	tec_has_orbital_food_deposit = {
		OR = {
			has_resource = food
			AND = {
				OR = {
					is_planet_class = pc_gas_giant
					is_planet_class = pc_frozen
				}
				NOR = {
					has_deposit_for = shipclass_research_station
					has_deposit_for = shipclass_mining_station
				}
			}
			inline_script = {
				script = mod_support/tec_trigger_include
				trigger = tec_has_orbital_food_deposit
			}
		}
	}
