tec_can_assemble_fractal_species = {
	has_trait = trait_tec_robot_fractal
	planet = {
		OR = {
			NOT = { exists = assembling_species }
			AND = {
				exists = assembling_species
				assembling_species = {
					OR = {
						has_trait = trait_tec_robot_fractal
						is_robotic = yes
					}
				}
			}
		}
	}
}
