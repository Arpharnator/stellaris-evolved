#trade bonus for ecu
col_ecu_trade = {
	icon = GFX_colony_type_habitat_trade

	potential = {
		OR = {
			uses_district_set = city_world
			tec_is_pd_ecu_with_trade = yes
		}
	}

	planet_modifier = {
		planet_structures_upkeep_mult = -0.10
		trade_value_mult = 0.25
	}

	weight_modifier = {
		modifier = {
			add = @stickiness
			has_designation = col_ecu_trade
		}
		add = value:housing_districts_value|FACTOR|2|
		complex_trigger_modifier = {
			trigger = num_districts
			parameters = {
				type = district_tec_arcology_trade
			}
			mode = add
			mult = 10
		}
		complex_trigger_modifier = {
			trigger = num_assigned_jobs
			parameters = {
				job = merchant # merchant jobs act as a pretty decent proxy for "this is a trade planet"
			}
			mode = add
			mult = 2
		}
	}
}
