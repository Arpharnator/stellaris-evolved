# pop_categories/inline_evolved_pop_upkeep_basics

# Resource Upkeep
	# Food
		upkeep = {
			trigger = {
				$trigger$

				has_food_upkeep = yes
				is_phototrophic = no
				tec_is_lithotrophic = no
			}
			# 1
			food = @living_standard_food_normal
		}
		upkeep = {	# Organic x trophic not combo
			trigger = {
				$trigger$

				has_food_upkeep = yes
				OR = {
					is_phototrophic = yes
					AND = {
						is_phototrophic = no
						tec_is_lithotrophic = yes
					}
				}
			}
			# 0.5
			food = @living_standard_phototrophic_normal
		}
		upkeep = {	# Organic combotrophic
			trigger = {
				$trigger$

				has_food_upkeep = yes
				is_phototrophic = yes
				tec_is_lithotrophic = yes
			}
			# 0.25
			food = @tec_lithotrophic_low
		}
	# /

	# Minerals
		upkeep = {
			trigger = {
				$trigger$

				has_mineral_upkeep = yes
				is_phototrophic = no
				tec_is_lithotrophic = no
			}
			# 1
			minerals = @living_standard_food_normal
		}
		upkeep = {
			trigger = {
				$trigger$

				has_mineral_upkeep = yes
				OR = {
					is_phototrophic = yes
					AND = {
						tec_is_lithotrophic = no
						is_phototrophic = no
					}
				}
			}
			# 0.5
			minerals = @living_standard_phototrophic_normal
		}
		upkeep = {
			trigger = {
				$trigger$

				has_mineral_upkeep = yes
				is_phototrophic = yes
				tec_is_lithotrophic = yes
			}
			# 0.25
			minerals = @tec_lithotrophic_low
		}
	# /
