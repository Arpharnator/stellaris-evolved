# modifiers/_tec_amenities

inline_script = {
	script = tec_type_hint
	trigger_block = "$trigger$"
	float = $count$
	number_or_value = "$mult$"
	bool = $default$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			AND = {
				exists = planet
				planet = {
					exists = owner

					$trigger$

					owner = {
						is_regular_empire = yes
					}
				}
			}
		}
	}
	planet_amenities_add = $count$
	mult = $mult$
}

triggered_planet_modifier = {
	potential = {
		exists = planet
		planet = {
			exists = owner

			$trigger$

			owner = {
				is_gestalt = yes
			}
		}
	}
	planet_amenities_no_happiness_add = $count$
	mult = $mult$
}
