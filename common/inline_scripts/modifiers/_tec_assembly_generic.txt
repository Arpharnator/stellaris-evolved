# modifiers/_tec_assembly_generic

inline_script = {
	script = tec_type_hint
	trigger_block = "$trigger$"
	number_or_value = "$mult$"
	bool = $default$
}

triggered_planet_modifier = {
	potential = {
		OR = {
			AND = {
				is_scope_valid = no
				always = $default$
			}
			planet = {
				exists = owner

				$trigger$

				OR = {
					NOT = { exists = assembling_species }
					assembling_species = {
						is_robotic = yes
					}
				}

			}
		}

	}
	modifier = {
		planet_pop_assembly_add = 1
	}
	mult = $mult$
}
