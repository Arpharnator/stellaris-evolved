# Robotic
	#inline_script = {
	#	script = modifiers/tec_assembly_automod
	#	robotic = yes
	#	amount = x
	#}
inline_script = {
	script = conditional/tec_equal
	value = $robotic$
	equal = yes
	code = "
		triggered_planet_modifier = {
			potential = {
				exists = owner
				owner = {
					OR = {
						has_origin = origin_mechanists
						is_individual_machine = yes
						has_country_flag = synthetic_empire
					}
				}
				any_owned_species = {
					has_auto_modding_trait = yes
				}
			}
			modifier = {
				auto_mod_monthly_add = $amount$
			}
		}
	"
}
