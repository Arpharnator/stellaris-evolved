# inline_evolved_add_job_civic_hive

# Hive extra jobs from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

inline_script = { # Scavenger jobs
	script = jobs/inline_evolved_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = {
			is_scavenger = yes
		}
	"
	job = tec_scavenger_drone
	count = $count$
	mult = $mult$
	display = $display$
}

inline_script = {
	script = mod_support/inline_evolved_inlines_include
	inlcude_script = jobs/inline_evolved_add_job_civic_hive
	parameters = "
		count = $count$
		mult = $mult$
		display = $display$
		trigger = \"$trigger$\"
	"
}
