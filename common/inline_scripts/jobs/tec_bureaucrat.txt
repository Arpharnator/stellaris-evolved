# jobs/tec_bureaucrat

inline_script = {	# bureaucrat
	script = jobs/tec_job
	trigger = "
		exists = owner

		$trigger$ 

		owner = {
			inline_script = { script = tec_read_cache scope = country trigger = tec_uses_regular_empire_jobs }
			is_megacorp = no
			tec_is_aiauth_empire = no
			tec_is_worker_coop_empire = no
		}
	"
	job = bureaucrat
	count = $count$
	mult = $mult$
	display = $display$
	default_job = $default_job$
	default_job = no
	default_trigger = "$default_trigger$"
	default_trigger = "always = no"
}
