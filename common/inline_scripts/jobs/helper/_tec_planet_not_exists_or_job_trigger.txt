# jobs/helper/tec_planet_not_exists_or_job_trigger.txt

inline_script = {
	script = tec_type_hint
	scripted_trigger = $trigger$
	bool = $negate$
}
OR = {
	NOT = { exists = planet }
	planet = {
		inline_script = {
			script = conditional/tec_equal_not
			value = $negate$
			equal = no
			code = "
				NOT = { inline_script = { script = tec_read_cache scope = planet trigger = $trigger$ } }
			"
		}

		inline_script = {
			script = conditional/tec_equal
			value = $negate$
			equal = no
			code = "
				inline_script = { script = tec_read_cache scope = planet trigger = $trigger$ }
			"
		}
	}
}
