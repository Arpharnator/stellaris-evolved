# jobs/tec_job_leader_background_planet

# Job without description, usually used when a job might be given by multiple triggered modifiers

inline_script = {
	script = conditional/tec_equal
	value = $default_job$
	equal = $job$
	code = "
		triggered_background_planet_modifier = {
			potential = {
				OR = {
					is_scope_valid = no
					AND = {
						exists = owner
						owner = {
							$default_trigger$
						}
					}
				}
			}
			job_$job$_add = $count$
			mult = $mult$
		}
	"
}

inline_script = {
	script = conditional/tec_number_not
	value = @[ | $count$ | ]
	equal = 0
	code = "
		triggered_background_planet_modifier = {
			potential = {
				exists = from
				from = {
					$trigger$
				}
			}
			job_$job$_add = $count$

			mult = $mult$
		}
	"
}
