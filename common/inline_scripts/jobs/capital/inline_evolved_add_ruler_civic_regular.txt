# Ruler jobs from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

inline_script = {
	script = jobs/inline_evolved_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = {
			has_valid_civic = civic_tec_regular_merchants
		}
	"
	job = merchant
	count = $count$
	mult = $mult$
	display = $display$
}
