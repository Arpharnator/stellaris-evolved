# jobs/resources/inline_evolved_bio_assembly_upkeep

upkeep = {
	trigger = {
		exists = owner
		OR = {
			AND = {
				NOT = { exists = assembling_species }
				owner = { is_lithoid = no }
			}
			AND = {
				exists = assembling_species
				assembling_species = {
					is_lithoid = no
				}
			}
		}
	}
	food = $base$
	mult = $mult$
}
upkeep = {
	trigger = {
		OR = {
			AND = {
				NOT = { exists = assembling_species }
				owner = { is_lithoid = yes }
			}
			AND = {
				exists = assembling_species
				assembling_species = {
					is_lithoid = yes
				}
			}
		}
	}
	minerals = $base$
	mult = $mult$
}
