# jobs/resources/inline_evolved_robotic_assembly_upkeep

upkeep = {
	trigger = {
		planet = {
			OR = {
				NOT = { exists = assembling_species }
				tec_allows_machine_assembly_for_cyborg = no
				assembling_species = {
					tec_is_cyborg_species = no
				}
			}
		}
	}
	alloys = $base$
	mult = $mult$
}

	trigger = {

				is_lithoid = no
			}
		}
	}
	alloys = @[$base$ / 2]
	food = @[$base$ / 2 * 5]
	mult = $mult$
}
