# jobs/tec_job_with_desc

# Job with description, usually used by default

inline_script = {
	script = conditional/tec_equal
	value = $default_job$
	equal = $job$
	code = "
		triggered_planet_modifier = {
			potential = {
				is_scope_valid = no
			}
			job_$job$_add = $count$
			mult = $mult$
		}

		triggered_desc = {
			trigger = {
				is_scope_valid = no
			}
			text = job_$job$_effect_desc
		}
	"
}

inline_script = {
	script = conditional/tec_number_not
	value = @[ | $count$ | ]
	equal = 0
	code = "
		triggered_planet_modifier = {
			potential = {
				$trigger$
			}
			job_$job$_add = $count$
			mult = $mult$
		}

		triggered_desc = {
			trigger = {
				$trigger$
				check_variable_arithmetic = {
					which = $mult$
					value > 0
				}
			}
			text = job_$job$_effect_desc
		}
	"
}
