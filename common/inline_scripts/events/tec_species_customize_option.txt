	script = tec_type_hint
	event = $event_loc$
	country_event = $event$
	scalar = $type$
	event_target = $target$
	portrait = $portrait$
	trigger = $trigger$
}

	name = $event_loc$.$type$
	trigger = {
		$trigger$ = yes
	}

	hidden_effect = {

		}

			id = $event$
		}
	}
}
