# districts/_tec_rural_district_modifiers

inline_script = {
	script = tec_type_hint
	float = $housing_scale$
	float = $building_slot_scale$
	float = $amenities_scale$
	menial_type = $type$
	trigger_block = "$trigger$"
	script_value = $mult_value$
}

# Housing
	inline_script = {
		script = conditional/tec_number_not
		value = $housing_scale$
		equal = 0
		code = "
			triggered_planet_modifier = {	# Scaled housing
				potential = {
					exists = owner
					$trigger$
				}
				planet_housing_add = $housing_scale$
				mult = value:tec_rural_housing|TYPE|$type$|MULT_VALUE|$mult_value$|
			}
		"
	}
# /

# Building slots
	inline_script = {
		script = conditional/tec_number_not
		value = $building_slot_scale$
		equal = 0
		code = "
			triggered_planet_modifier = {	# Scaled housing
				potential = {
					exists = owner
					$trigger$
				}
				planet_max_buildings_add = $building_slot_scale$
				mult = value:tec_rural_building_slots|TYPE|$type$|
			}
		"
	}
# /

# Building slots
	inline_script = {
		script = conditional/tec_number_not
		value = $amenities_scale$
		equal = 0
		code = "
			inline_script = {
				script = modifiers/tec_amenities
				count = $amenities_scale$
				mult = value:tec_rural_amenities|TYPE|$type$|
				trigger = \\\"$trigger$\\\"
				default = no
			}
		"
	}
# /
