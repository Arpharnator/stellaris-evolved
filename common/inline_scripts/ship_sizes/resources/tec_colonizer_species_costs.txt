# Consumer goods for non-gestalt empires
# 200 normally
cost = {
    trigger = {
        from = {
            is_gestalt = no
            is_virtual_empire = no
            NOT = { has_active_tradition = tr_tec_outreach_final_3 }
        }
    }
    consumer_goods = @colony_cost_base
}
