ai_resource_production = {
	trigger = {
		exists = owner
		owner = {
			NOR = {
				has_deficit = minerals
				has_deficit = energy
				has_deficit = food
				has_deficit = consumer_goods
			}
		}
		num_unemployed > 0
		OR = {
			AND = {
				free_building_slots = 0
				free_district_slots = 0
			}
			num_unemployed > 5
		}
		$trigger$
		any_owned_pop = {
			is_unemployed = yes
			can_work_specific_job = $job$
		}
	}
	ai_unobtainium = 1000
}
