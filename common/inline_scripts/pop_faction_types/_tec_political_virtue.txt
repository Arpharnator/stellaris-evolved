produces = {	# Political virtue
	trigger = {
		exists = owner
		owner = {
			has_valid_civic = civic_tec_regular_political_virtue
		}
		faction_approval >= 0.5
	}
	$resource_1$ = $count_1$
	inline_script = {
	    script = conditional/tec_equal_not
	    value = $resource_2$
	    equal = no

			$resource_2$ = $count_2$
	    "
	}
	inline_script = {
		script = conditional/tec_equal_not
		value = $resource_2$
		equal = no
		code = "
			$resource_3$ = $count_3$
	    "
	}
}
