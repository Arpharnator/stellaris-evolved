namespace = plant

# Gaia Seed building chain completed.
planet_event = {
	id = plant.100
	is_triggered_only = yes
	title = "plant.100.name"
	desc = "plant.100.desc"
	picture = GFX_evt_gaia

	trigger = {
		last_building_changed = building_gaiaseeders_4
		NOT = {
			is_planet_class = pc_gaia
		}
	}

	immediate = {
		# Turn planet into Gaia world.
		# Hehe was faster than you for adding pd compat roadd!
			if = {
				limit = {
					has_global_flag = planetarydiversity
				}
				if = {
					limit = { 
						is_pd_tidally_locked = yes
					}
					change_pc = pc_tidallygaia
				}
				if = {
					limit = { 
						is_pd_superhab = yes
					}
					change_pc = pc_gaia_superhab
				}
				if = {
					limit = { 
						is_pd_cave = yes
					}
					change_pc = pc_gaia_cave
				}
				if = {
					limit = { 
						is_pd_tidally_locked = no
						is_pd_superhab = no
						is_pd_cave = no
					}
					change_pc = pc_gaia
				}
			}
			else = {
				change_pc = pc_gaia
			}
		# /

		# Remove Gaia Seed building.
		remove_building = building_gaiaseeders_4

		# Evolved
		# Adds the permanent building
		add_building = building_tec_gaiaseeders_gaia
	}

	option = {
		# Not Hive-mind
		name = plant.100.a
		trigger = {
			owner = { is_gestalt = no }
		}
		add_modifier = {
			modifier = celebrate_idyllic_bloom
			days = 180
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = 12
			}
		}
	}

	option = {
		name = plant.100.b
		# Hive-mind
		trigger = {
			owner = { is_hive_empire = yes }
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = 18
			}
		}
	}

	option = {
		name = plant.100.c
		# Hive-mind
		trigger = {
			owner = { is_machine_empire = yes }
		}
		owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = 18
			}
		}
	}

	after = {
		tooltip = {
			change_pc = pc_gaia
			remove_building = building_gaiaseeders_4
			add_building = building_tec_gaiaseeders_gaia
		}
	}
}

country_event = {
	id = plant.105
	is_triggered_only = yes
	title = "plant.100.name"
	desc = "plant.105.desc"
	picture = GFX_evt_gaia

	trigger = {
		from.planet = {
			NOR = {
				is_planet_class = pc_gaia
				has_planet_flag = anti_cheese_idyllic_bloom_flag
			}
			has_holding = {
				holding = holding_idyllic_bloom
				owner = root
			}
		}
	}

	option = {
		name = plant.100.a
		trigger = {
			is_machine_empire = no
		}
		add_monthly_resource_mult = {
			resource = unity
			value = 12
		}
		hidden_effect = {
			from.owner = {
				country_event = { id = plant.110 }
			}
			if = {
				limit = {
					from.planet = {
						is_planet_class = pc_ocean
						owner = {
							has_ascension_perk = ap_hydrocentric
						}
					}
				}

							has_ascension_perk = ap_hydrocentric

						add_modifier = { 
							modifier =  terraformed_our_ocean_world
							years = 3
						}
					}
				}
			}
		}
	}

	after = {
		from.planet = {
			if = {
				limit = { 
					is_pd_tidally_locked = yes
				}
				change_pc = pc_tidallygaia
			}

					is_pd_superhab = no

				}
				change_pc = pc_gaia
			}
			remove_holding = {
				holding = holding_idyllic_bloom
				owner = root
			}
			add_holding = {
				holding = holding_tec_idyllic_bloom_gaia
