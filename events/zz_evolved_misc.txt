namespace = tec_misc

planet_event = { # Ecoworld remove modifier
	id = tec_misc.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		tec_is_ecoworld = yes
		OR = {
			tec_is_arcology = yes
			tec_invalidates_ecoworld_project_comp = yes
		}
	}

	immediate = {
		remove_modifier = pm_tec_ecoworld_planet
	}
	
}

namespace = tec_caching

event = { # General caching, run once 
	id = tec_caching.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_galaxy_planet = {
			limit = {
				OR = {
					is_colony = yes
					is_colonizable = yes
				}
			}
			tec_cache_planet_triggers = yes
			if = {
				limit = {
					is_colony = yes
				}
				tec_cache_capital_triggers = yes
			}
		}
		every_galaxy_pop = {
			tec_cache_shackled_triggers = yes
		}
	}
}
